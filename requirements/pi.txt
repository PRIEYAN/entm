# Raspberry Pi (64-bit / aarch64) — MarianMT CTranslate2 int8 inference.
# No torch. No IndicTransToolkit.
#
# Pinning policy: exact-pin only the packages that affect model correctness
# (the runtime, the tokenizer stack). The HTTP layer is floored, not pinned —
# over-pinning it caused a real failure on DietPi/Python 3.13, where an exact
# pin with no matching aarch64 wheel triggered a source build that aborted the
# whole install (pip is atomic: one build failure installs nothing).
#
# Verified on: laptop x86_64/py3.11 and DietPi aarch64/py3.13.

# --- model-critical: exact pins -------------------------------------------
# ctranslate2 4.8.1 needs transformers >=4.56 for the `dtype` kwarg.
# 4.57.6 is the version this model was converted and verified with.
ctranslate2==4.8.1
transformers==4.57.6
# 0.2.1, not 0.2.0: 0.2.0 ships no aarch64 wheel for Python 3.13, so pip falls
# back to a source build needing cmake + pkg-config (absent on stock DietPi).
sentencepiece==0.2.1

# --- supporting: floors, so wheels resolve on any board/Python ------------
sacremoses>=0.1.1
protobuf>=5.29
numpy>=2.0
fastapi>=0.115
uvicorn[standard]>=0.34
